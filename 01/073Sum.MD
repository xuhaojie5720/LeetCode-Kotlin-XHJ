# 3Sum（三数之和）—Kotlin 学习样本

本页并列展示 Java 与 Kotlin 的等价实现，逻辑与复杂度完全一致，便于迁移对照。

## 原始 Java 代码

```java
import java.util.*;

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> ret = new ArrayList<>();
        Arrays.sort(nums);

        for (int k = nums.length - 1; k >= 2;) {
            int i = 0;
            int j = k - 1;

            while (i < j) {
                int sum = nums[i] + nums[j] + nums[k];
                if (sum == 0) {
                    List<Integer> list = new ArrayList<>();
                    list.add(nums[i]);
                    list.add(nums[j]);
                    list.add(nums[k]);
                    ret.add(list);

                    for (; (j > i) && (nums[j - 1] == nums[j]); j--) ;
                    j--;

                    for (; (i < j) && (nums[i + 1] == nums[i]); i++) ;
                    i++;
                } else if (sum > 0) {
                    for (; (j > i) && (nums[j - 1] == nums[j]); j--) ;
                    j--;
                } else {
                    for (; (i < j) && (nums[i + 1] == nums[i]); i++) ;
                    i++;
                }
            }

            for (; (k >= 2) && (nums[k - 1] == nums[k]); k--) ;
            k--;
        }

        return ret;
    }
}
```

## 等价 Kotlin 代码（逻辑不变）

```kotlin
class Solution {
    fun threeSum(nums: IntArray): List<List<Int>> {
        val res = mutableListOf<List<Int>>() // Kotlin：可变列表收集结果
        nums.sort() // Kotlin：IntArray 原地排序

        var k = nums.lastIndex // Kotlin：lastIndex 代替 length-1
        while (k >= 2) {
            var i = 0
            var j = k - 1
            while (i < j) {
                val sum = nums[i] + nums[j] + nums[k]
                when {
                    sum == 0 -> {
                        res.add(listOf(nums[i], nums[j], nums[k])) // Kotlin：构建只读三元组

                        while (j > i && nums[j - 1] == nums[j]) j-- // 去重 j
                        j--

                        while (i < j && nums[i + 1] == nums[i]) i++ // 去重 i
                        i++
                    }
                    sum > 0 -> {
                        while (j > i && nums[j - 1] == nums[j]) j-- // 去重 j
                        j--
                    }
                    else -> { // sum < 0
                        while (i < j && nums[i + 1] == nums[i]) i++ // 去重 i
                        i++
                    }
                }
            }

            while (k >= 2 && nums[k - 1] == nums[k]) k-- // 去重 k
            k--
        }
        return res
    }
}
```

## Kotlin 要点对照

- 原地排序：`IntArray.sort()`；空间 O(1)。
- 容器选择：`mutableListOf` 收集结果，子列表用只读 `listOf`。
- 末尾索引：`lastIndex` 提升可读性，避免 `size-1`。
- 分支表达：`when` 代替多层 if-else，可读性更好。
- 去重策略：与 Java 一致的 while 推进，保持 O(n^2)。
